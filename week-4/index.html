Selectors

id selector

// html
<div id="this-one"></div>

// js
$('#this-one')

class selector

// html
<div class="those"></div>
<div class="those"></div>
<div class="those"></div>

// js
$('.those')

tag selector

// html
<div class="those"></div>
<div class="those"></div>
<div class="those"></div>

// js
$('div')

還有很多種 selector，不過跟 css 一樣先不教

可以參考這裡

jQuery Object

$ 這個 function 回傳的都是 jQuery Object

它是個像是 array 但又不是 array 的東西

底下代表的是 selector 選取到的 0 個, 1 個或多個東西

沒選到東西時


// js
var noThisOne = $('.no-this-one');
noThisOne.length // 0
noThisOne[0] // undefined

選到 1 個東西時

// html
<div class="only-one"></div>
// js
var onlyOne = $('.only-one');
noThisOne.length // 1
noThisOne[0] // 選到的第一個 DOM 物件

選到 1 個以上的東西時

// html
<div class="not-only-one"></div>
<div class="not-only-one"></div>
<div class="not-only-one"></div>
// js
var notOnlyOne = $('.not-only-one');
notOnlyOne.length // 3
notOnlyOne[0] // 選到的第一個 DOM 物件
notOnlyOne[1] // 選到的第二個 DOM 物件
notOnlyOne[2] // 選到的第三個 DOM 物件

可以看到他跟 array 有一些相似的地方

而用 $ 變成 jQuery object 後就會有後面要講的那些超能力

Traversing

處理樹狀圖的位置關係、過濾

.children()

取得他的下一層

// html
<div class="wrapper">
	<div class="target"></div>
	<div class="target"></div>
	<div class="target"></div>
</div>

// js
var wrapper = $('.wrapper');
wrapper.children().length; // 3

.children(selector)

取得他的下一層符合條件的

// html
<div class="wrapper">
	<div class="target"></div>
	<div class="not-target"></div>
	<div class="target"></div>
</div>

// js
var wrapper = $('.wrapper');
wrapper.children('.target').length; // 2

.parent()

取得他的上一層

// html
<div class="target">
	<div class="inner"></div>
</div>

// js
var inner = $('.inner');
var parent = inner.parent()
parent.length // 1
parent.attr('class'); // target

.siblings()

取得他的同一層其他人不包括自己

// html
<div class="wrapper">
	<div class="selector"></div>
	<div class="target"></div>
	<div class="target"></div>
</div>

// js
var selector = $('.selector');
var siblings = selector.siblings()
siblings.length // 2

.find(selector)

取得他的裡面任何一個符合條件的

// html
<div class="wrapper">
	

		<div class="target"></div>
		

			<div class="target"></div>
		

		<div class="target"></div>
	

</div>

// js
var wrapper = $('.wrapper');
wrapper.find('.target').length; // 3

.closest(selector)

取得他的位置往外找最快符合條件的那個

// html
<div class="wrapper">
	

		<div></div>
		

			<div class="inner"></div>
		

		<div></div>
	

</div>

// js
var inner = $('.inner');
inner.closest('.target') //

.filter()

濾出符合條件的

// html
<div class="wrapper">
	<div></div>
	<div class="target"></div>
	<div class="target"></div>
</div>

// js
var wrapper = $('.wrapper');
var target = wrapper.children().filter('.target');
target.length // 2

.first()

取回的裡面第一個

// html
<div class="wrapper">
	<div class="target"></div>
	<div></div>
	<div></div>
</div>

// js
var wrapper = $('.wrapper');
var target = wrapper.children().first();
target.attr('class') // target

.last()

取回的裡面最後一個

// html
<div class="wrapper">
	<div></div>
	<div></div>
	<div class="target"></div>
</div>

// js
var wrapper = $('.wrapper');
var target = wrapper.children().last();
target.attr('class') // target

.eq(index)

取回 index 位置的 (跟 array 一樣的排序方式)

// html
<div class="wrapper">
	<div></div>
	<div class="target"></div>
	<div></div>
</div>

// js
var wrapper = $('.wrapper');
var target = wrapper.children().eq(1);
target.attr('class') // target

Manipulation

取得狀態和設定狀態

.addClass

加一個 class 上去

// html before
<div id="target"></div>

// js
var $target = $('#target');
$target.addClass('a-new-class');

// html after
<div id="target" class="a-new-class"></div>

.removeClass

移除一個 class

// html before
<div id="target" class="a-new-class"></div>

// js
var $target = $('#target');
$target.removeClass('a-new-class');

// html after
<div id="target"></div>

.hasClass

判斷有沒有 class

// html
<div id="target" class="a-new-class"></div>

// js
var $target = $('#target');
$target.hasClass('a-new-class') // true

.append

添加到裡面一層的最後面

// html before
<div class="wrapper">
	<div></div>
	<div></div>
</div>

// js
var $wrapper = $('.wrapper');
$wrapper.append('<div class="target"></div>');

// html after
<div class="wrapper">
	<div></div>
	<div></div>
	<div class="target"></div>
</div>

.prepend

添加到裡面一層的最前面

// html before
<div class="wrapper">
	<div></div>
	<div></div>
</div>

// js
var $wrapper = $('.wrapper');
$wrapper.prepend('<div class="target"></div>');

// html after
<div class="wrapper">
	<div class="target"></div>
	<div></div>
	<div></div>
</div>

.after

添加到後面

// html before
<div class="wrapper">
	<div></div>
	<div></div>
</div>

// js
var $wrapper = $('.wrapper');
$wrapper.after('<div class="target"></div>');

// html after
<div class="wrapper">
	<div></div>
	<div></div>
</div>
<div class="target"></div>

.before

添加到前面

// html before
<div class="wrapper">
	<div></div>
	<div></div>
</div>

// js
var $wrapper = $('.wrapper');
$wrapper.before('<div class="target"></div>');

// html after
<div class="target"></div>
<div class="wrapper">
	<div></div>
	<div></div>
</div>

.text

取得裡面的字串

// html
<div class="wrapper">裡面的字</div>

// js
var $wrapper = $('.wrapper');
$wrapper.text() // '裡面的字'

.text(text)

改寫裡面的字串

// html before
<div class="wrapper">裡面的字</div>

// js
var $wrapper = $('.wrapper');
$wrapper.text('想要別的字串');

<div class="wrapper">想要別的字串</div>

.remove

移除

// html before
<div>
	<div class="wrapper">
		<div></div>
		<div></div>
	</div>
</div>
// js
var $wrapper = $('.wrapper');
$wrapper.remove();

// html after
<div>
</div>

.attr

取得屬性

// html before
<div id="target" class="wrapper">
	<div></div>
	<div></div>
</div>

// js
var $wrapper = $('.wrapper');
$wrapper.attr('id') // target

.attr(attrName, attrValue)

設定屬性

// html before
<div id="target" class="wrapper">
	<div></div>
	<div></div>
</div>

// js
var $wrapper = $('.wrapper');
$wrapper.attr('id', 'another'); //

// html after
<div id="another" class="wrapper">
	<div></div>
	<div></div>
</div>

Events

監聽事件的發生並去處理它

名稱 發生處理狀況
click 點擊的時候
dblclick 連點兩次的時候
focus 開始在輸入框輸入時候
blur 離開輸入框時候
change 輸入值改變的後
keydown 鍵盤按下去的時候
keyup 鍵盤按下去的鍵上來的時候
mouseenter 滑鼠進到元素裡面的時候
mouseleave 滑鼠移出元素的時候

重點 function 只有一個就是 .on(eventName, handler)

// html
<button id="target-btn">
	是個按鈕
</button>

// js
var $btn = $('.btn');
$btn.on('click', function() {
	// ..做一些事，任何事，你希望按鈕按了要幹麻？
	alert('你希望按鈕按了要幹麻？');
});

Effects

一些動畫效果

這部分比較難所以只講一個範例

// html
<div id="effect" style="position: relative; left: 0; background-color: black; width: 200px; height: 200px;"></div>

// js
var $effect = $('#effect');
var left = +($effect.css('left').split('px')[0]);
$effect.animate({
	left: '+=800'
}, 2000);

剛剛那是個比較底層的 function 可以做比較多設定，
例如：效果時間、速度曲線、完成後要做的事情

純粹想快速做效果的話：.fadeIn(), .fadeOut(), .show(), .hide(), .slideDown(), .slideUp() 好用許多

可以參考這裡

瀏覽器一打開畫面時，DOM 那時候即時還沒好

所以寫 DOM 操作的程式時，必須等到 DOM ready

通常你應該把這類程式碼放到 document.ready

尤其是如果你的 js 載入時機比較早或是不固定時


$(document).ready(function() {
	// ..你的程式碼
});
